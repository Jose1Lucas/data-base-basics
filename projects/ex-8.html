-- Criar a tabela de clientes
    CREATE TABLE clientes (
        id SERIAL PRIMARY KEY,
        nome VARCHAR(100) NOT NULL,
        email VARCHAR(100) UNIQUE NOT NULL,
        data_cadastro DATE NOT NULL
    );
    
    -- Inserir alguns dados de exemplo
    INSERT INTO clientes (nome, email, data_cadastro) VALUES
        ('João Silva', 'joao@email.com', '2025-02-19'),
        ('Maria Oliveira', 'maria@email.com', '2025-02-19'),
        ('Carlos Santos', 'carlos@email.com', '2025-02-20');
    
    -- Criar a função para contar clientes cadastrados em um dia específico
    CREATE OR REPLACE FUNCTION contar_clientes_por_dia(data_especifica DATE) 
    RETURNS INT AS $$
    DECLARE
        total_clientes INT;
    BEGIN
        SELECT COUNT(*) INTO total_clientes
        FROM clientes
        WHERE data_cadastro = data_especifica;
        
        RETURN total_clientes;
    END;
    $$ LANGUAGE plpgsql;
    
    -- Exemplo de uso da função
    SELECT contar_clientes_por_dia('2025-02-19');
